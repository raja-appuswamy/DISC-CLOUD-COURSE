%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame {\frametitle{Cost of IT}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{itemize}
	\item When you are using IT there are three primary costs associated with it:
	\begin{itemize}
		\item Software cost (Media + License cost/user)
		\item Support cost (vendor support, updates, ...)
		\item Management cost (Manpower, IT infrastructure, ...)
	\end{itemize}
\end{itemize}		
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame {\frametitle{Traditional model}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{itemize}
	\item a.k.a Classic model
	\item Software provider develops software and charges a license fee per user for the client
	\item The provider may charge a support fee /user
	\item The management of the software is the clients responsibility
	\begin{itemize}
		\item Up to 4x the cost of the actual software per year!
		\item Infrastructure, Manpower, software maintenance
	\end{itemize}
	\item Traditional Software example: Oracle, SQL Server, Outlook, ...
\end{itemize}
}

\frame {\frametitle{Open Source Model}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{itemize}
	\item a.k.a ``Free'' model
	\item Software provider packages Open Source Software and provides it at little or no cost to the client
	\item The provider makes money on support, charges a higher fee than traditional model
	\item The cost of Managing the software remains the same as Traditional Model
	\begin{itemize}
		\item Up to 4x the cost of the actual software per year
		\item Infrastructure, Manpower, software maintenance
	\end{itemize}
	\item Traditional Software example: Oracle, SQL Server, Outlook, ...
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame {\frametitle{Outsourcing Model}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{itemize}
	\item Primary cost of Software Management is in Manpower
	\item Why not delegate the management of software to a country with cheaper labor costs
	\item Outsource the management of software for a flat fee – keep IT management costs under control
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame {\frametitle{Hybrid and Hybrid+ models}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{itemize}
	\item Business Software Requirements do not change often.
	\begin{itemize}
		\item ERP, Financials, CRM etc.
	\end{itemize}
	\item Why reinvent the wheel? Standardize, Specialize and Repeat
	\begin{itemize}
		\item Create a flexible version of the Software that can be quickly configured and deployed.
		\item Automate support through remote access.
	\end{itemize}
	\item Sell easy to deploy software to many clients.
	\begin{itemize}
		\item Decrease the Margin
		\item Increase the Customers
	\end{itemize}
	\item Hybrid+ is more advanced – charge a flat monthly fee for the software, support and management
\end{itemize}
}
