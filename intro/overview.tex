\frame {\frametitle{This Course}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{itemize}
	\item \textbf{This course is about the fundamentals of cloud computing}
    \begin{itemize}
		\item Infrastructure (systems and algorithms)
		\item Cloud applications and programming models
    \end{itemize}

	\item \textbf{What you will learn(roadmap)}
	\begin{itemize}  
		\item \textbf{In-depth description of Hadoop MapReduce}
		\begin{itemize}
		  \item Architecture internals
		  \item Cluster deployments
		\end{itemize}

		\item \textbf{In-depth description of Apache Spark}
		\begin{itemize}
		  \item Architecture internals
		\end{itemize}

		\item \textbf{Relational Algebra and High-Level Languages}
		\begin{itemize}
		  \item Basic operators and their equivalence in MapReduce
		  \item Apache SparkSQL
		\end{itemize}

	\end{itemize}  

\end{itemize}
}
